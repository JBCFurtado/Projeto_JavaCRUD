CREATE DATABASE CRUD;
USE CRUD;

CREATE TABLE PROFESSOR(MATRICULA BIGINT AUTO_INCREMENT,
NOME VARCHAR (255),
RG VARCHAR (255),
CPF VARCHAR(255),
ENDERECO VARCHAR (255),
BAIRRO VARCHAR (255),
TELEFONE VARCHAR(255),
EMAIL VARCHAR(255),
SEXO VARCHAR(255),
DISCIPLINA VARCHAR (255),
ATUACAO VARCHAR (255),
PRIMARY KEY (MATRICULA));

CREATE TABLE ALUNO(MATRICULA BIGINT AUTO_INCREMENT,
NOME VARCHAR (255),
RG VARCHAR (255),
CPF VARCHAR(255),
ENDERECO VARCHAR (255),
BAIRRO VARCHAR (255),
TELEFONE VARCHAR(255),
SEXO VARCHAR(255),
CURSO VARCHAR (255),
PERIODO VARCHAR (255),
PRIMARY KEY (MATRICULA));
